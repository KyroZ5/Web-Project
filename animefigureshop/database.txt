--cart contents


CREATE TABLE cartContents (
    email VARCHAR(50) NOT NULL,
    prodCode VARCHAR(15),
    PRIMARY KEY (email, prodCode),
    FOREIGN KEY (email) REFERENCES accounts(email)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (prodCode) REFERENCES productDetails(prodCode)
        ON DELETE CASCADE
        ON UPDATE CASCADE
) ENGINE=InnoDB;	
ALTER TABLE cartContents
ADD amount INT NOT NULL DEFAULT 1;	





CREATE TABLE Orders (
    orderID INT AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(50) NOT NULL,
    paymentMethod ENUM('Bank','GcashPaymaya','Paypal','Cash') NOT NULL,
    logisticsMethod ENUM('Delivery','Pickup') NOT NULL,
    orderDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (email) REFERENCES accounts(email)
);
ALTER TABLE Orders
ADD COLUMN orderStatus ENUM('Pending','Approved','Rejected') NOT NULL DEFAULT 'Pending';
CREATE TABLE OrderItems (
    orderItemID INT AUTO_INCREMENT PRIMARY KEY,
    orderID INT NOT NULL,
    prodCode VARCHAR(15) NOT NULL,
    quantity INT NOT NULL,
    FOREIGN KEY (orderID) REFERENCES Orders(orderID),
    FOREIGN KEY (prodCode) REFERENCES ProductDetails(prodCode)
);
CREATE TABLE DeliveryDetails (
    deliveryID INT AUTO_INCREMENT PRIMARY KEY,
    orderID INT NOT NULL,
    address VARCHAR(255) NOT NULL,
    apartmentSuite VARCHAR(255), -- optional
    city VARCHAR(100) NOT NULL,
    province VARCHAR(100) NOT NULL,
    zipCode VARCHAR(20) NOT NULL,
    phoneNumber VARCHAR(20) NOT NULL,
    FOREIGN KEY (orderID) REFERENCES Orders(orderID)
);
